function getRangeData(min, max, step, prefix, label) {
 if (min > max) return false;
 var ret = [[0, label]];
 if (step < 0) {
   for (var i = max; i >= min; i += step)
     ret.push([i, prefix + i]);
 } else if (step > 0) {
   for (var i = min; i <= max; i += step)
     ret.push([i, prefix + i]);
 }
 return ret;
}
